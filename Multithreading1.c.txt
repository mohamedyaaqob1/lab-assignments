void main()
{
  Summing implements Runnable {

    int a;

     Summing(int a) {
        this.a = a;
    }

     void run() {
        addRondom();
    }

     void addRondom() {
        Random rand = new Random();
        int n = rand.nextInt(10) + 1;
        printif("number generated: " + n);
        synchronized (this) {
            a += n;
        }
    }
}
 static void main(String[] args) {
    int base = 0;
    Summing sum2 = new Summing(base);
